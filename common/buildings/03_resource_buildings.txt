building_hydroponics_farm = {
	base_buildtime = @b1_time

	category = resource

	potential = {
		exists = owner
		owner = {
			OR = {
				country_uses_food = yes
				is_ai = no
			}
		}
		NOT = { is_planet_class = pc_machine }
		NOT = { has_modifier = resort_colony }
	}

		buildings_simple_allow = yes
		str_hydroponics_allow = yes

		exists = owner
		OR = {
			str_hydroponics_destroy = yes

				is_ai = yes
				country_uses_food = no
			}
			planet = {
				NOT = { is_planet_class = pc_habitat }
				free_building_slots = 0
				owner = {
					is_ai = yes

						resource = food

			}

		potential = {
			exists = owner

						is_anglers_empire = yes

			}
		}

			job_farmer_add = @b1_jobs
		}

		potential = {
			exists = owner
			owner = { is_fallen_empire_spiritualist = yes }
		}
		modifier = {
			job_fe_acolyte_farm_add = @b1_jobs
		}

		potential = {
			exists = owner

			job_farmer_add = 1

			job_agri_drone_add = @b1_jobs

			exists = owner

				is_regular_empire = yes
				is_fallen_empire_spiritualist = no

building_bio_reactor = {
	base_buildtime = 240

	category = resource

	potential = {
		exists = owner
		owner = {
			OR = {
				is_machine_empire = yes
				is_mechanical_empire = yes
				is_lithoid_empire = yes
			}
			OR = {
				owner = { is_ai = no }
				owner = { country_uses_food = no }
			}
		}
	}

	allow = {
		hidden_trigger = {
			owner = {
				OR = {
					is_ai = no
					AND = {
						is_ai = yes
						NOT = {
							has_deficit = food
							has_country_flag = buildings_new_reactor_building
						}
						resource_stockpile_compare = {
							resource = food
							value = 5000
						}
					}
				}
			}
		}
	}

	destroy_trigger = {
		exists = owner
		owner = {
			is_ai = yes
			has_deficit = food
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 200
		}
		produces = {
			trigger = {
				exists = owner
				owner = { NOT = { has_deficit = food } }
			}
			energy = 20
		}
		upkeep = {
			food = 25
		}
	}

	prerequisites = {
		"tech_bio_reactor"
	}

	on_queued = {
		IF = {
			limit = {
				owner = { has_technology = tech_construction_templates }
			}
			owner = { set_timed_country_flag = { flag = buildings_new_reactor_building days = 270 } }
		}
		ELSE_IF = {
			limit = {
				owner = { has_technology = tech_assembly_pattern }
			}
			owner = { set_timed_country_flag = { flag = buildings_new_reactor_building days = 390 } }
		}
		ELSE = {
			planet = {
				owner = { set_timed_country_flag = { flag = buildings_new_reactor_building days = 510 } }
			}
		}
	}
}

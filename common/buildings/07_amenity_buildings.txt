building_luxury_residence = {
	base_buildtime = @b1_time

	category = amenity

	potential = {
		OR = {
			NOT = { is_planet_class = pc_habitat }
			has_planet_flag = advanced_habitat
		}
		NOT = { has_modifier = slave_colony }
		exists = owner
		owner = {
			is_regular_empire = yes
			NOT = { has_valid_civic = civic_shared_burden }
		}
		NOT = { has_building = building_low_tech_capital }
	}

	allow = {
		hidden_trigger = {
			OR = {
				owner = { is_ai = no }
				AND = {
					NOR = {
						has_building = building_luxury_residence
						has_building = building_research_lab_1
					}
					is_capital = no
					num_pops >= 13
					free_jobs > 2
					OR = {
						AND = {
							free_amenities < -4
							free_amenities > -9
							OR = {
								free_district_slots < 3
								num_pops < 16
							}
						}
						AND = {
							free_building_slots > 1
							free_amenities < -2
							free_amenities > -6
							free_district_slots = 0
							has_building = building_holo_theatres
						}
					}
				}
			}
		}
	}

	destroy_trigger = {
		exists = owner
		OR = {
			AND = {
				free_building_slots = 0
				free_jobs < -2
				free_housing > 10
				free_district_slots = 0
				owner = {
					is_ai = yes
				}
			}
			AND = {
				owner = { lategame = no }
				num_buildings = {
					type = building_research_lab_1
					value > 1	
				}
				free_jobs < 1
				free_amenities > 6

				}
			}

		}

		planet_housing_add = 3

			}

			planet_housing_add = 1

		NOT = {	has_modifier = slave_colony	}

		OR = {
			owner = {
				NOT = { is_ai = yes }
			}
			AND = {
				NOT = { has_available_jobs = entertainer }
				NOT = { has_forbidden_jobs = entertainer }
			}
		}
		NOT = { has_building = building_low_tech_capital }

			OR = {

					free_amenities < -5

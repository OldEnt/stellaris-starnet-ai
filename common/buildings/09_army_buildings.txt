building_military_academy = {
	base_buildtime = @b2_time
	empire_limit = {
		base = 1
	}
	can_be_ruined = no
	category = army

	allow = {
		exists = owner
		owner = { is_ai = no }
		has_upgraded_capital = yes
		if = {
			limit = {
				exists = owner
				owner = { is_ai = yes }
			}
			OR = {
				has_building = building_stronghold
				has_building = building_fortress
			}
		}
	}

	country_modifier = {
		country_commander_cap_add = 1
		country_naval_cap_mult = 0.05
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
		}
		modifier = {
			job_politician_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_evaluator_add = 2
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b2_minerals
			volatile_motes = @b2_rare_cost
		}
		upkeep = {
			energy = @b2_upkeep
			volatile_motes = @b2_rare_upkeep
		}
	}

	prerequisites = {
		"tech_centralized_command"
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_politician_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_evaluator_effect_desc
	}
}

building_stronghold = {
	base_buildtime = 240
	position_priority = 100
	can_be_ruined = no
	category = army
	exempt_from_ai_planet_specialization = no

	potential = {
		buildings_simple_allow = yes
	}

	allow = {
		OR = {
			owner = { is_ai = no }
			AND = {
				NOR = {
					has_building = building_stronghold
					has_building = building_fortress
				}
				is_capital = no
				num_buildings = {
					type = any
					value > 5
				}
			}
		}
	}

	destroy_trigger = {
		exists = owner
		owner = { is_ai = yes }
		AND = {
			OR = {
				num_buildings = { type = building_stronghold value > 1 }
				has_building = building_fortress
			}
			free_building_slots = 0
			owner = {
				used_naval_capacity_percent < 0.6
				is_at_war = no # don't destroy while at war
			}
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
		}
		modifier = {
			job_soldier_add = @b1_jobs
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_warrior_drone_add = @b1_jobs
		}
	}

	resources = {
		category = planet_buildings_strongholds
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = 1
		}
	}

	prerequisites = {
		"tech_planetary_defenses"
	}

	upgrades = {
		"building_fortress"
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_warrior_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_soldier_effect_desc
	}
}

building_planetary_shield_generator = {
	base_buildtime = @b3_time
	base_cap_amount = 1
	position_priority = 100
	can_be_ruined = no
	category = army

	allow = {
		has_upgraded_capital = yes
		hidden_trigger = { owner = { is_ai = no } }
		if = {
			limit = {
				exists = orbital_defence
			}
			orbital_defence = {
				exists = starbase
				starbase = {
					NOR = {
						has_starbase_building = ring_orbital_shield_generator
						is_starbase_building_building = ring_orbital_shield_generator
					}
				}
			}
		}
	}

	destroy_trigger = {
		exists = owner
		owner = { is_ai = yes }
	}

	planet_modifier = {
		planet_orbital_bombardment_damage = -0.5
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b2_minerals
		}
		upkeep = {
			energy = @b2_upkeep
		}
	}

	prerequisites = {
		"tech_planetary_shield_generator"
	}
}

	potential = {
	}
	allow = {
	}
	effect = {
		hidden_effect = {
			event_target:global_event_country = {
				switch = {
					trigger = has_country_flag
					str_register_ship_size_str_var_ai_mode_option_1 = {
						set_variable = {
							which = str_register_ship_size_str_var_ai_mode
							value = -1
						}
						remove_country_flag = str_register_ship_size_str_var_ai_mode_option_1
						set_country_flag = str_register_ship_size_str_var_ai_mode_option_2
					}
					str_register_ship_size_str_var_ai_mode_option_2 = {
						set_variable = {
							which = str_register_ship_size_str_var_ai_mode
							value = 0
						}
						remove_country_flag = str_register_ship_size_str_var_ai_mode_option_2
						set_country_flag = str_register_ship_size_str_var_ai_mode_option_3
					}
					str_register_ship_size_str_var_ai_mode_option_3 = {
						clear_variable = str_register_ship_size_str_var_ai_mode
						remove_country_flag = str_register_ship_size_str_var_ai_mode_option_3
					}
					default = {
						set_country_flag = str_register_ship_size_str_var_ai_mode_option_1
						set_variable = {
							which = str_register_ship_size_str_var_ai_mode
							value = 1
						}
					}
				}
				if = {
					limit = {
						is_variable_set = str_register_ship_size_str_var_ai_mode
						is_variable_set = str_register_ship_size_str_var_ai_value
					}
					set_variable = {
						which = str_register_ship_size_str_var_ai_value_input
						value = str_register_ship_size_str_var_ai_value
					}
					multiply_variable = {
						which = str_register_ship_size_str_var_ai_value_input
						value = str_register_ship_size_str_var_ai_mode
					}
				}
			}
		}
	}
}

	}

					}

								which = str_register_ship_size_str_var_ai_value
								value = 5
							}
						}
						set_variable = {
							which = str_register_ship_size_str_var_ai_value
							value = 10
						}
					}
					else_if = {
						limit = {
							check_variable = {
								which = str_register_ship_size_str_var_ai_value
								value = 10
							}
						}
						set_variable = {
							which = str_register_ship_size_str_var_ai_value
							value = 100
						}
					}
					else_if = {
						limit = {
							check_variable = {
								which = str_register_ship_size_str_var_ai_value
								value = 100
							}
						}
						set_variable = {
							which = str_register_ship_size_str_var_ai_value
							value = 0.1
						}
					}
					else = {
						set_variable = {
							which = str_register_ship_size_str_var_ai_value
							value = 5
						}
					}
				}
				if = {
					limit = {
						NOT = {
							is_variable_set = str_register_ship_size_str_var_ai_value
						}
					}
					set_variable = {
						which = str_register_ship_size_str_var_ai_value
						value = 5
					}
				}
				set_variable = {
					which = str_register_ship_size_str_var_ai_value_input
					value = str_register_ship_size_str_var_ai_value
				}
				if = {
					limit = {
						is_variable_set = str_register_ship_size_str_var_ai_mode

						which = str_register_ship_size_str_var_ai_value_input
						value = str_register_ship_size_str_var_ai_mode
					}
				}
			}

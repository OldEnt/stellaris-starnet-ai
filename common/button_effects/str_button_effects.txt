str_var_ai_will_not_declare_war_until_reset = {
	potential = {
	}
	allow = {
	}
	effect = {
		hidden_effect = {
			event_target:global_event_country = {
				if = {
					limit = {
						is_variable_set = str_var_ai_will_not_declare_war_until
					}
					clear_variable = str_var_ai_will_not_declare_war_until
				}
			}
			str_calculate_real_truce_years = yes
		}
	}
}

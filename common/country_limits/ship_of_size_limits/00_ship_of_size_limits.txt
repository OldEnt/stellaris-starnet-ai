progenitor_ships_limit = {
	ship_types = { 
		offspring_corvette 
		offspring_destroyer
		offspring_cruiser
		offspring_battleship
	}
	
	base = 1
	max = 200
	naval_cap_fraction = 0.1
	
	show = {
		is_scope_valid = yes
		has_origin = origin_progenitor_hive
	}
}

titan_ships_limit = { 
	ship_types = { 
		titan
	}
	
	base = value:str_value_limits_base_titan	# 16 # 1 * titan's size_multiplier
	max = value:str_value_limits_max_titan	# 320 # 20 * 16
	naval_cap_fraction = value:str_value_limits_naval_cap_fraction_titan	# 0.08
	
	show = {
		is_scope_valid = yes
		has_technology = tech_titans
	}
}

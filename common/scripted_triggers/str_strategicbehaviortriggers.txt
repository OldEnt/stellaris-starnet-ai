str_prepare_for_war = {
	years_passed > 12
	#Alchemy code used to see if there is any country WE are hostile towards
	#The reason we even ask this question is the following:
	#If ai behavior is already hostile (I can't mod the behaviors sadly)
	#that means ai will claim and declare war as soon as it has the influence
	#so saving influence won't be wasted
	OR = {
		is_at_war = yes
		any_neighbor_country = {
			OR = {
				is_country_type = default
				merg_is_default_empire = yes
				has_ascension_perk = ap_become_the_crisis
			}
			prev = {
				OR = {
					is_hostile_to = prev
					is_unfriendly_to = prev
				}
				# relative_power = {
				# 	who = prev
				# 	category = all
				# 	value >= equivalent
				# }
			}
		}
	}
	resource_stockpile_compare = {
		resource = minerals
		value > 1500
	}
	resource_income_compare = {
		resource = minerals
		value > 40
	}
	str_raw_ecocrisis_level_0 = no
	and = {
		used_naval_capacity_integer > 25
		or = {
			has_ascension_perk = ap_interstellar_dominion
			and = {
				has_ascension_perk = ap_executive_vigor
				has_edict = capacity_subsidies
			}
			and = {
				num_ascension_perks > 1
				has_tradition = tr_supremacy_adopt
			}
		}
		resource_stockpile_compare = {
			resource = alloys
			value > 1000
		}
	}
}

str_strategic_stop_building_ships = {
	optimize_memory
	AND = {
		resource_stockpile_compare = {
			resource = $UPKEEP_RESOURCE_NAME$			# Check stockpile of this upkeep resource.
			value < $UPKEEP_RESOURCE_STOCKPILE_COMPARE$			# Stockpile value.
		}
		resource_income_compare = {
			resource = $UPKEEP_RESOURCE_NAME$
			value < $UPKEEP_RESOURCE_INCOME_COMPARE$			# Income value.
		}
		AND = {
			resource_income_to_expenditure_balance_ratio = {
				resource = $UPKEEP_RESOURCE_NAME$
				value < $UPKEEP_RESOURCE_INCOME_TO_EXPENDITURE_BALANCE_RATIO$				# Inflow to outflow ratio.
			}
		}
	}
}
